import javax.swing.*;
import java.awt.*;
import java.awt.event.*;
import java.sql.*;

 class CarLoginScreen extends JFrame {

    // --- PROJEYİ BAŞLATAN MAIN METODU ---
    public static void main(String[] args) {
        try {
            // İşletim sistemi görünümünü yükle (Daha net yazı tipleri için)
            UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
        } catch (Exception e) {}

        SwingUtilities.invokeLater(() -> new CarLoginScreen().setVisible(true));
    }
    // ------------------------------------

    private static final Color BG_COLOR = new Color(18, 30, 49);
    private static final Color FIELD_BG_COLOR = new Color(40, 53, 75);
    private static final Color TEXT_COLOR = Color.WHITE;
    private static final Color PLACEHOLDER_COLOR = Color.LIGHT_GRAY;
    private static final Color BLUE_BUTTON_COLOR = new Color(0, 123, 255);

    private JTextField userField;
    private JPasswordField passField;
    private JButton loginButton;

    public CarLoginScreen() {
        setTitle("39 Park | Giriş Yap");
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setSize(500, 700);
        setLocationRelativeTo(null);
        setResizable(false);

        JPanel mainPanel = new JPanel(new GridBagLayout());
        mainPanel.setBackground(BG_COLOR);
        GridBagConstraints gbc = new GridBagConstraints();
        gbc.insets = new Insets(10, 40, 10, 40);
        gbc.fill = GridBagConstraints.HORIZONTAL;
        gbc.gridx = 0;

        // --- BAŞLIK ---
        JLabel titleLabel = new JLabel("39 PARK GİRİŞ");
        titleLabel.setFont(new Font("Segoe UI", Font.BOLD, 36));
        titleLabel.setForeground(TEXT_COLOR);
        titleLabel.setHorizontalAlignment(SwingConstants.CENTER);
        gbc.gridy = 0;
        gbc.insets.bottom = 30;
        mainPanel.add(titleLabel, gbc);

        // --- INPUTLAR ---
        userField = new JTextField("Plaka Gir");
        styleTextField(userField, "Plaka Gir");
        gbc.gridy = 1;
        mainPanel.add(userField, gbc);

        passField = new JPasswordField("Şifre Gir");
        stylePasswordField(passField, "Şifre Gir");
        gbc.gridy = 2;
        mainPanel.add(passField, gbc);

        // --- BUTON ---
        loginButton = new JButton("Giriş Yap");
        loginButton.setFont(new Font("Segoe UI", Font.BOLD, 18));
        loginButton.setBackground(BLUE_BUTTON_COLOR);
        loginButton.setForeground(TEXT_COLOR);
        loginButton.setFocusPainted(false);
        loginButton.setBorderPainted(false);
        loginButton.setCursor(new Cursor(Cursor.HAND_CURSOR));
        loginButton.setPreferredSize(new Dimension(100, 50));
        gbc.gridy = 3;
        gbc.insets.top = 20;
        mainPanel.add(loginButton, gbc);

        // *** BUTON AKSİYONU ***
        loginButton.addActionListener(e -> {
            String plaka = userField.getText();
            String sifre = new String(passField.getPassword());

            if (plaka.equals("Plaka Gir") || sifre.equals("Şifre Gir")) {
                JOptionPane.showMessageDialog(this, "Lütfen bilgileri girin!");
                return;
            }

            if (login(plaka, sifre)) {
                JOptionPane.showMessageDialog(this, "Giriş Başarılı! Hoş geldin: " + plaka);
                this.dispose();
                new ParkUI(plaka).setVisible(true); // ParkUI ekranını aç
            } else {
                JOptionPane.showMessageDialog(this, "Hatalı Plaka veya Şifre!", "Hata", JOptionPane.ERROR_MESSAGE);
            }
        });

        add(mainPanel);
    }

    // DB Bağlantısı
    public Connection baglan() {
        try {
            String url = "jdbc:sqlserver://localhost:1433;databaseName=AkilliOtoparkDB;encrypt=false;trustServerCertificate=true;";
            String user = "sa";
            String pass = "12345sa";
            return DriverManager.getConnection(url, user, pass);
        } catch (Exception e) {
            e.printStackTrace();
            return null;
        }
    }

    public boolean login(String plaka, String sifre) {
        String sql = "SELECT 1 FROM Araclar WHERE plaka=? AND sifre=?";
        try (Connection con = baglan()) {
            if (con == null) return false;
            try (PreparedStatement ps = con.prepareStatement(sql)) {
                ps.setString(1, plaka);
                ps.setString(2, sifre);
                ResultSet rs = ps.executeQuery();
                return rs.next();
            }
        } catch (Exception e) {
            e.printStackTrace();
            return false;
        }
    }

    // --- Stil Metotları ---
    private void styleTextField(JTextField field, String placeholder) {
        field.setBackground(FIELD_BG_COLOR);
        field.setForeground(PLACEHOLDER_COLOR);
        field.setFont(new Font("Segoe UI", Font.PLAIN, 16));
        field.setBorder(BorderFactory.createCompoundBorder(
                BorderFactory.createLineBorder(FIELD_BG_COLOR, 2),
                BorderFactory.createEmptyBorder(10, 15, 10, 15)));

        field.addFocusListener(new FocusAdapter() {
            public void focusGained(FocusEvent e) {
                if (field.getText().equals(placeholder)) {
                    field.setText("");
                    field.setForeground(TEXT_COLOR);
                }
            }
            public void focusLost(FocusEvent e) {
                if (field.getText().isEmpty()) {
                    field.setForeground(PLACEHOLDER_COLOR);
                    field.setText(placeholder);
                }
            }
        });
    }

    private void stylePasswordField(JPasswordField field, String placeholder) {
        field.setBackground(FIELD_BG_COLOR);
        field.setForeground(PLACEHOLDER_COLOR);
        field.setFont(new Font("Segoe UI", Font.PLAIN, 16));
        field.setEchoChar((char) 0);
        field.setBorder(BorderFactory.createCompoundBorder(
                BorderFactory.createLineBorder(FIELD_BG_COLOR, 2),
                BorderFactory.createEmptyBorder(10, 15, 10, 15)));

        field.addFocusListener(new FocusAdapter() {
            public void focusGained(FocusEvent e) {
                String pass = new String(field.getPassword());
                if (pass.equals(placeholder)) {
                    field.setText("");
                    field.setEchoChar('•');
                    field.setForeground(TEXT_COLOR);
                }
            }
            public void focusLost(FocusEvent e) {
                String pass = new String(field.getPassword());
                if (pass.isEmpty()) {
                    field.setForeground(PLACEHOLDER_COLOR);
                    field.setText(placeholder);
                    field.setEchoChar((char) 0);
                }
            }
        });
    }
}
